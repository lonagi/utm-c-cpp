def warshall(M):
    n = M.shape[0]
    W = M
    for k in range(n):
        for i in range(n):
            for j in range(n):
                W[i, j] = W[i, j] or (W[i, k] and W[k, j])
    return W

def findParents(D,v):
    for i in D.edges():
        if(i[1]==v):
            return i[0]
        
def findChildren(D,v):
    ch =[]
    for i in D.edges():
        if(i[0]==v):
            ch.append(i[1])
    return ch

def findBrothers(D,v):
    ch = []
    for i in D.edges():
        if(i[1]==v and i[0] in get1col(D.edges()) and i[1] != v):
            ch.append(i[0])
    return ch
